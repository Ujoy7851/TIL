def solution(bridge_length, weight, truck_weights):
    answer = 0
    bridge = [0] * bridge_length
    total_weight = 0
    n = len(truck_weights)
    i = 0
    passed = 0

    while True:
        if passed == n:
            break

        t = bridge.pop(0)
        total_weight -= t
        if t != 0:
            passed += 1

        if i < n:
            truck = truck_weights[i]
            if total_weight + truck <= weight:
                bridge.append(truck)
                i += 1
                total_weight += truck
            else:
                bridge.append(0)
        else:
            bridge.append(0)

        answer += 1
        # print(answer, bridge)

    return answer
